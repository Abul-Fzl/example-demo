void init_serial (void)
{
PINSEL0 = 0x00000005; // Enable RxD0 and TxD0
U1LCR =0x00000083;//#8 bits, no Parity, 1 Stop bit
U1DLL = 0x000000C3;//1/9600 Baud Rate @ 30MHz VPB Clock
U1LCR =0x00000003;
}
